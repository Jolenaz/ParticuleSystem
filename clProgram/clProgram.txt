__kernel void drawPoints(__global float4* buffer, uint caracSize) 
{
    float4 ret;
    uint index = get_global_id(0);
    uint tmp = caracSize * caracSize;
    ret.s2 = (float)(index / tmp) / caracSize;
    ret.s0 = (float)((index % tmp) / caracSize) / caracSize;
    ret.s1 = (float)((index % tmp) % caracSize) / caracSize;
    ret.s3 = 1.0f;
    buffer[index] = ret;
}
